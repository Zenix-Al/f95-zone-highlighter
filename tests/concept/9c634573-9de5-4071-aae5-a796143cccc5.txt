// ==UserScript==
// @name         F95Zone Data Siphon (Test Sandbox)
// @namespace    http://tampermonkey.net/
// @version      1.0
// @description  Standalone sandbox to verify zero-latency XHR JSON siphoning on the latest threads page
// @match        *://f95zone.to/*
// @run-at       document-start
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    console.log("🐒 [Sandbox] Micro-trap injected at document-start. Waiting for connection...");

    // Initialize the global bucket
    window.__f95RawData = null;

    // Preserve original native XMLHttpRequests
    const originalOpen = XMLHttpRequest.prototype.open;
    const originalSend = XMLHttpRequest.prototype.send;

    // Trap the request configuration to grab the target URL
    XMLHttpRequest.prototype.open = function(method, url, ...args) {
        this._interceptUrl = url;
        return originalOpen.apply(this, [method, url, ...args]);
    };

    // Trap the network payload execution
    XMLHttpRequest.prototype.send = function(...args) {
        this.addEventListener('readystatechange', function() {
            // Wait until request finishes (readyState 4) and verify target endpoint
            if (this.readyState === 4 && this._interceptUrl && this._interceptUrl.includes('latest_data.php')) {
                try {
                    const json = JSON.parse(this.responseText);

                    if (json && json.status === "ok" && json.msg?.data) {
                        // Siphon the clean, raw data array into memory
                        window.__f95RawData = json.msg.data;

                        console.log("⚡ [Sandbox] SUCCESS! Siphoned raw network payload array instantly:");
                        console.dir(window.__f95RawData);
                    }
                } catch (e) {
                    console.error("❌ [Sandbox] Failed to parse intercepted JSON string:", e);
                }
            }
        });
        return originalSend.apply(this, args);
    };

    // -------------------------------------------------------------
    // VERIFICATION POLLING: Simulates what your mega script sees at document-idle
    // -------------------------------------------------------------
    const checkBucket = setInterval(() => {
        if (document.readyState === 'complete' || document.readyState === 'interactive') {
            clearInterval(checkBucket);

            console.log("📦 [Sandbox] DOM is fully loaded. Examining global data memory bucket...");
            if (window.__f95RawData) {
                console.log("✅ [Sandbox] Verification Complete! Clean data array is waiting in `window.__f95RawData`:", window.__f95RawData);
                console.log(`📊 [Sandbox] Total thread records ready for layout extraction without DOM scraping: ${window.__f95RawData.length}`);
            } else {
                console.warn("⚠️ [Sandbox] DOM loaded, but `window.__f95RawData` is empty. Confirm you are on the /latest tab or that the page executed an initial network fetch.");
            }
        }
    }, 200);

})();
